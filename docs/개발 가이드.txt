        user = fields.ForeignKeyField(

        hospital_name = fields.CharField(max_length=200, null=True)

    from pydantic import BaseModel, Field


    class GuideCreateDTO(BaseModel):
        """가이드 생성 요청"""
        title: str | None = Field(default=None, max_length=200)
        hospital_name: str | None = Field(default=None, max_length=200)
        # ... 필요한 필드 ...


    class GuideResponseDTO(BaseModel):
        """가이드 응답"""
        guide_id: int
        title: str | None = None
        hospital_name: str | None = None
        created_at: datetime

        class Config:
            from_attributes = True  # ⚠️ 반드시 추가 (Tortoise 모델 → DTO 변환용)

Step 4: 서비스 만들기 (app/services/)
    # app/services/guide_service.py (새로 생성)

    import logging
    from fastapi import HTTPException, status
    from app.models.guide import HealthGuide
    from app.models.user import User

    logger = logging.getLogger(__name__)


    async def create_guide(user: User, body) -> HealthGuide:
        """가이드 생성"""
        guide = await HealthGuide.create(
            user=user,
            title=body.title,
            hospital_name=body.hospital_name,
        )
        logger.info(f"가이드 생성: guide_id={guide.guide_id}, user_id={user.user_id}")
        return guide


    async def get_user_guides(user: User) -> list[HealthGuide]:
        """사용자의 가이드 목록 조회"""
        return await HealthGuide.filter(user=user).order_by("-created_at")


    async def get_guide_by_id(user: User, guide_id: int) -> HealthGuide:
        """가이드 상세 조회 (본인 것만)"""
        guide = await HealthGuide.get_or_none(guide_id=guide_id, user=user)
        if guide is None:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="가이드를 찾을 수 없습니다.",
            )
        return guide

Step 5: API 만들기 (app/apis/v1/)
    # app/apis/v1/guide.py

    from fastapi import APIRouter, Depends, status

    from app.core.dependencies import get_current_user  # ⭐ 인증 의존성
    from app.dtos.common_dto import ResponseDTO          # ⭐ 공통 응답 래퍼
    from app.dtos.guide_dto import GuideCreateDTO, GuideResponseDTO
    from app.models.user import User
    from app.services import guide_service

    router = APIRouter()  # ⚠️ 변수명은 반드시 "router"


    @router.post(
        "/",
        response_model=ResponseDTO[GuideResponseDTO],
        status_code=status.HTTP_201_CREATED,
        summary="가이드 생성",
    )
    async def create_guide(
        body: GuideCreateDTO,
        user: User = Depends(get_current_user),  # ⭐ 인증된 사용자 자동 주입
    ):
        """새 건강 가이드를 생성합니다."""
        guide = await guide_service.create_guide(user, body)
        return ResponseDTO(
            success=True,
            message="가이드 생성 성공",
            data=GuideResponseDTO.model_validate(guide),
        )


    @router.get(
        "/",
        response_model=ResponseDTO[list[GuideResponseDTO]],
        summary="가이드 목록 조회",
    )
    async def list_guides(user: User = Depends(get_current_user)):
        """현재 사용자의 가이드 목록을 조회합니다."""
        guides = await guide_service.get_user_guides(user)
        data = [GuideResponseDTO.model_validate(g) for g in guides]
        return ResponseDTO(success=True, data=data)

    class DocAnalysisJob(Model):

        #      status_code VARCHAR(20) DEFAULT 'JOB_PENDING'
        #      FOREIGN KEY (status_grp, status_code) → common_code

        # Tortoise: 문자열 필드로 저장, 서비스에서 검증
        status_grp = fields.CharField(max_length=20, default="JOB_STATUS")
        status_code = fields.CharField(max_length=20, default="JOB_PENDING")

8. S3 파일 업로드/다운로드
8-1. API에서 파일 업로드

    from fastapi import UploadFile, File
    from app.core.s3 import upload_file, generate_s3_key

    @router.post("/upload")
    async def upload_document(
        file: UploadFile = File(...),
        user: User = Depends(get_current_user),
    ):
        # 1. S3 키 생성 (충돌 방지 UUID 포함)
        s3_key = generate_s3_key(
            folder="medical-docs",           # 폴더명 (자기 기능에 맞게)
            filename=file.filename or "doc",
            user_id=user.user_id,
        )
        # 결과 예: "medical-docs/2026/03/31/user_1/a1b2c3d4_report.pdf"

        # 2. S3에 업로드
        await upload_file(
            file=file.file,
            s3_key=s3_key,

        user: User = Depends(get_current_user),
    ):

        )

        # 2. task_id를 DB에 저장하고 응답
        return ResponseDTO(
            success=True,
            data={"task_id": task_id, "status": "pending"},
        )

15. 자주 발생하는 실수와 해결법
#"Table not found" 에러
    원인: 모델을 만들었지만 MODELS 리스트에 등록하지 않음
    해결: app/db/databases.py의 MODELS에 "app.models.내모델" 추가

#"No TortoiseContext" 에러
    원인: DB 초기화 전에 모델 쿼리를 실행
    해결: 모델 쿼리는 반드시 API 핸들러나 서비스 함수 안에서만 실행
        (모듈 최상위 레벨에서 실행하면 안 됨)

#"router not found" 또는 API가 Swagger에 안 나옴
    원인: apis/v1/__init__.py에 라우터가 등록되지 않음
    해결: 이미 등록되어 있으므로 router 변수명이 "router"인지 확인
        → 파일에 "router = APIRouter()" 가 반드시 있어야 함

#DTO 변환 에러 ("model_validate failed")
    원인: DTO의 Config에 from_attributes = True 가 없음
    해결: class Config:
            from_attributes = True

#Git 충돌
    원인: 다른 팀원과 같은 파일을 수정
    해결: 자기 담당 파일만 수정하면 충돌이 거의 없습니다.
        공통 파일(models/__init__.py, databases.py) 수정이 필요하면 팀장에게 요청.

#Import 에러
    원인: 서비스 파일을 새로 만들었는데 __init__.py에 등록 안 함
    해결: services/ 에는 __init__.py 등록이 필요 없습니다.
        직접 import하면 됩니다: from app.services.guide_service import ...
